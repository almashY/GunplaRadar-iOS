classDiagram

        <<View>>

    class StockDiffRegistrationView {
        <<View>>

    class RestockCalendarViewModel {
        <<ViewModel>>
        -calendarDays: [CalendarDay]
        -selectedDate: Date?
        -errorMessage: String?
        +loadCalendar(year: Int, month: Int)
        +selectDate(_ date: Date)
    }

    class StockDiffViewModel {
        <<ViewModel>>
        -restockDate: Date
        -targetItem: GunplaItem

    class StockDelayRecord {
        <<Model>>
        +id: UUID
        +storeId: UUID
        +itemId: UUID
        +restockDate: Date
        +actualStockDate: Date
        +delayHours: Double
    }

    class GunplaItemRepository {
        <<protocol>>
        +fetchAll() throws [GunplaItem]
        +fetch(byRestockDate date: Date) throws [GunplaItem]

    class PatrolPlanRepository {
        <<protocol>>
        +fetchAll() throws [PatrolPlan]
    }

    class StockDelayRecordRepository {
        <<protocol>>
        +save(_ record: StockDelayRecord) throws
        +fetch(by storeId: UUID) throws [StockDelayRecord]
    }

    class StoreRepository {
        <<protocol>>
        +fetch(by id: UUID) throws Store?
        +save(_ store: Store) throws
    }

    class StockDelayRecordRepositoryImpl {
        -context: ModelContext
